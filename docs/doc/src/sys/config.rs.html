50/// 設定データをロードする。
51pub fn load() -> Result<()> {
52    {
53        // デフォルト設定ファイルを削除する
54        info!("remove {}", CONFIG_DEF_FILE);
55        if let Err(e) = remove_file(CONFIG_DEF_FILE) {
56            warn!(
57                "removing {} failed (the first time execution?): {}",
58                CONFIG_DEF_FILE, e
59            );
60        }
61        // デフォルト設定を書き出す
62        // permission=600 でアトミックに必ず新規作成する、失敗したらエラー
63        info!("writing default config to {}", CONFIG_DEF_FILE);
64        let main_cfg: Config = Default::default();
65        let main_toml = toml::to_string(&main_cfg)?;
66        let mut f = OpenOptions::new()
67            .write(true)
68            .create_new(true)
69            .mode(0o600)
70            .open(CONFIG_DEF_FILE)
71            .with_context(|| format!("Failed to open {CONFIG_DEF_FILE}"))?;
72        f.write_all(main_toml.as_bytes())
73            .with_context(|| format!("Failed to write {CONFIG_DEF_FILE}"))?;
74        info!("OK: written to {}", CONFIG_DEF_FILE);
75        // close
76    }
77
78    let toml_str = {
79        // 設定ファイルを読む
80        // open 後パーミッションを確認し、危険ならエラーとする
81        info!("loading config: {}", CONFIG_FILE);
82        let mut f = OpenOptions::new()
83            .read(true)
84            .open(CONFIG_FILE)
85            .with_context(|| format!("Failed to open {CONFIG_FILE} (the first execution?)"))
86            .with_context(|| {
87                format!("HINT: Copy {CONFIG_DEF_FILE} to {CONFIG_FILE} and try again")
88            })?;
89
90        let metadata = f.metadata()?;
91        let permissions = metadata.permissions();
92        let masked = permissions.mode() & 0o777;
93        ensure!(
94            masked == 0o600,
95            "Config file permission is not 600: {:03o}",
96            permissions.mode()
97        );
98
99        let mut toml_str = String::new();
100        f.read_to_string(&mut toml_str)
101            .with_context(|| format!("Failed to read {CONFIG_FILE}"))?;
102        info!("OK: {} loaded", CONFIG_FILE);
103
104        toml_str
105        // close f
106    };
107
108    // グローバル変数に設定する
109    let mut config = CONFIG.write().unwrap();
110    *config = Some(toml::from_str(&toml_str)?);
111
112    {
113        // 現在設定ファイルを削除する
114        info!("remove {}", CONFIG_CUR_FILE);
115        if let Err(e) = remove_file(CONFIG_CUR_FILE) {
116            warn!(
117                "removing {} failed (the first time execution?): {}",
118                CONFIG_CUR_FILE, e
119            );
120        }
121        // 現在設定を書き出す
122        // permission=600 でアトミックに必ず新規作成する、失敗したらエラー
123        info!("writing current config to {}", CONFIG_CUR_FILE);
124        let main_toml = toml::to_string(&*config)?;
125        let mut f = OpenOptions::new()
126            .write(true)
127            .create_new(true)
128            .mode(0o600)
129            .open(CONFIG_CUR_FILE)
130            .with_context(|| format!("Failed to open {CONFIG_CUR_FILE}"))?;
131        f.write_all(main_toml.as_bytes())
132            .with_context(|| format!("Failed to write {CONFIG_CUR_FILE}"))?;
133        info!("OK: written to {}", CONFIG_CUR_FILE);
134        // close
135    }
136
137    Ok(())
138}

140pub struct ConfigGuard;
141
142impl Drop for ConfigGuard {
143    fn drop(&mut self) {
144        unset();
145    }
146}
147
148#[must_use]
149pub fn set(cfg: Config) -> ConfigGuard {
150    let mut config = CONFIG.write().unwrap();
151    assert!(config.is_none());
152    *config = Some(cfg);
153    ConfigGuard
154}
155
156pub fn unset() {
157    let mut config = CONFIG.write().unwrap();
158    assert!(config.is_some());
159    *config = None;
160}
161
162pub fn get<F, R>(f: F) -> R
163where
164    F: FnOnce(&Config) -> R,
165{
166    let config = CONFIG.read().unwrap();
167    f(config.as_ref().unwrap())
168}
