97impl Line {
98    /// コンストラクタ。
99    pub fn new() -> Result<Self> {
100        info!("[line] initialize");
101        let config = config::get(|cfg| cfg.line.clone());
102        let client = Client::builder().timeout(TIMEOUT).build()?;
103
104        Ok(Self {
105            config,
106            client,
107            chat_history: None,
108            chat_timeout: None,
109            func_table: None,
110        })
111    }
112
113    async fn init_openai(&mut self, ctrl: &Control) {
114        if self.chat_history.is_some() && self.func_table.is_some() {
115            return;
116        }
117
118        // トークン上限を算出
119        // Function 定義 + 前文 + (使用可能上限) + 出力
120        let (model_info, reserved) = {
121            let openai = ctrl.sysmods().openai.lock().await;
122
123            (
124                openai.model_info_offline(),
125                openai.get_output_reserved_token(),
126            )
127        };
128
129        let mut chat_history = ChatHistory::new(model_info.name);
130        assert!(chat_history.get_total_limit() == model_info.context_window);
131        let pre_token: usize = self
132            .config
133            .prompt
134            .instructions
135            .iter()
136            .map(|text| chat_history.token_count(text))
137            .sum();
138        let reserved = FUNCTION_TOKEN + pre_token + reserved;
139        chat_history.reserve_tokens(reserved);
140        info!("[line] OpenAI token limit");
141        info!("[line] {:6} total", model_info.context_window);
142        info!("[line] {reserved:6} reserved");
143        info!("[line] {:6} chat history", chat_history.usage().1);
144
145        let mut func_table = FunctionTable::new(Arc::clone(ctrl), Some("line"));
146        func_table.register_basic_functions();
147        register_draw_picture(&mut func_table);
148
149        let _ = self.chat_history.insert(chat_history);
150        let _ = self.func_table.insert(func_table);
151    }
152
153    pub async fn chat_history(&mut self, ctrl: &Control) -> &ChatHistory {
154        self.init_openai(ctrl).await;
155        self.chat_history.as_ref().unwrap()
156    }

309    /// <https://developers.line.biz/ja/reference/messaging-api/#send-reply-message>
310    ///
311    /// <https://developers.line.biz/ja/docs/messaging-api/text-character-count/>
312    ///
313    /// `texts` のそれぞれについて、長すぎるならば分割し、
314    /// 文字列メッセージ配列として送信する。
315    /// 配列の最大サイズは 5。
316    pub async fn reply_multi(&self, reply_token: &str, texts: &[&str]) -> Result<ReplyResp> {
317        let mut messages = Vec::new();
318        for text in texts {
319            ensure!(!text.is_empty(), "text must not be empty");
320            let splitted = split_message(text);
321            messages.extend(splitted.iter().map(|&chunk| Message::Text {
322                text: chunk.to_string(),
323            }));
324        }
325        ensure!(messages.len() <= 5, "text too long: {}", texts.len());
326
327        let req = ReplyReq {
328            reply_token: reply_token.to_string(),
329            messages,
330            notification_disabled: None,
331        };
332        let resp = self.post_auth_json(URL_REPLY, &req).await?;
333        info!("{:?}", resp);
334
335        Ok(resp)
336    }

358        };
359        let resp = self.post_auth_json(URL_PUSH, &req).await?;
360        info!("{:?}", resp);

362        Ok(resp)
363    }

384    /// レスポンスの内容を確認しながら json に変換する。
385    ///
386    /// * HTTP Status が成功の場合
387    ///   * Response body JSON を T に変換できればそれを返す。
388    ///   * 変換に失敗したらエラー。
389    /// * HTTP Status が失敗の場合
390    ///   * Response body JSON を [ErrorResp] にパースできればその [Debug] を
391    ///     メッセージとしてエラーを返す。
392    ///   * 変換に失敗した場合、JSON ソースをメッセージとしてエラーを返す。
393    async fn check_resp<'a, T>(resp: reqwest::Response) -> Result<T>
394    where
395        T: for<'de> Deserialize<'de>,
396    {
397        // https://qiita.com/kzee/items/d01e6f3e00dfadb9a00b
398        let status = resp.status();
399        let body = resp.text().await?;
400
401        if status.is_success() {
402            Ok(serde_json::from_reader::<_, T>(body.as_bytes())?)
403        } else {
404            match serde_json::from_str::<ErrorResp>(&body) {
405                Ok(obj) => bail!("{status}: {:?}", obj),
406                Err(json_err) => bail!("{status} - {json_err}: {body}"),
407            }
408        }
409    }
410
411    async fn get_auth_json<'a, T>(&self, url: &str) -> Result<T>
412    where
413        T: for<'de> Deserialize<'de>,
414    {
415        info!("[line] GET {url}");
416        let token = &self.config.token;
417        let resp = self
418            .client
419            .get(url)
420            .header("Authorization", format!("Bearer {token}"))
421            .send()
422            .await?;
423
424        Self::check_resp(resp).await
425    }
426
427    async fn post_auth_json<T, R>(&self, url: &str, body: &T) -> Result<R>
428    where
429        T: Serialize + Debug,
430        R: for<'de> Deserialize<'de>,
431    {
432        info!("[line] POST {url} {:?}", body);
433        let token = &self.config.token;
434        let resp = self
435            .client
436            .post(url)
437            .header("Authorization", format!("Bearer {token}"))
438            .json(body)
439            .send()
440            .await?;
441
442        Self::check_resp(resp).await
443    }

446fn split_message(text: &str) -> Vec<&str> {
447    // UTF-16 5000 文字で分割
448    let mut result = Vec::new();
449    // 左端
450    let mut s = 0;
451    // utf-16 文字数
452    let mut len = 0;
453    for (ind, c) in text.char_indices() {
454        // 1 or 2
455        let clen = c.len_utf16();
456        // 超えそうなら [s, ind) の部分文字列を出力
457        if len + clen > MSG_SPLIT_LEN {
458            result.push(&text[s..ind]);
459            s = ind;
460            len = 0;
461        }
462        len += clen;
463    }
464    if len > 0 {
465        result.push(&text[s..]);
466    }
467
468    result
469}

471fn register_draw_picture(func_table: &mut FunctionTable<FunctionContext>) {
472    let mut properties = HashMap::new();
473    properties.insert(
474        "keywords".to_string(),
475        ParameterElement {
476            type_: vec![ParameterType::String],
477            description: Some("Keywords for drawing. They must be in English.".to_string()),
478            ..Default::default()
479        },
480    );
481    func_table.register_function(
482        Function {
483            name: "draw".to_string(),
484            description: Some("Draw a picture".to_string()),
485            parameters: Parameters {
486                properties,
487                required: vec!["keywords".to_string()],
488                ..Default::default()
489            },
490            ..Default::default()
491        },
492        move |bctx, ctx, args| Box::pin(draw_picture(bctx, ctx, args)),
493    );
494}
495
496async fn draw_picture(
497    bctx: Arc<BasicContext>,
498    ctx: FunctionContext,
499    args: &FuncArgs,
500) -> Result<String> {
501    let keywords = function::get_arg_str(args, "keywords")?.to_string();
502
503    let ctrl = bctx.ctrl.clone();
504    taskserver::spawn_oneshot_fn(&ctrl, "line_draw_picture", async move {
505        let url = {
506            let mut ai = bctx.ctrl.sysmods().openai.lock().await;
507
508            ai.generate_image(&keywords, 1)
509                .await?
510                .pop()
511                .ok_or_else(|| anyhow!("parse error"))?
512        };
513        {
514            let line = bctx.ctrl.sysmods().line.lock().await;
515            line.push_image_message(&ctx.reply_to, &url).await?;
516        }
517        Ok(())
518    });
519
520    Ok("Accepted. The result will be automatially posted later. Assistant should not draw for now.".to_string())
521}

527    #[test]
528    fn split_long_message() {
529        let mut src = String::new();
530        assert!(split_message(&src).is_empty());
531
532        for i in 0..MSG_SPLIT_LEN {
533            let a = 'A' as u32;
534            src.push(char::from_u32(a + (i as u32 % 26)).unwrap());
535        }
536        let res = split_message(&src);
537        assert_eq!(1, res.len());
538        assert_eq!(src, res[0]);
539
540        src.push('0');
541        let res = split_message(&src);
542        assert_eq!(2, res.len());
543        assert_eq!(&src[..MSG_SPLIT_LEN], res[0]);
544        assert_eq!(&src[MSG_SPLIT_LEN..], res[1]);
545
546        src.pop();
547        src.pop();
548        src.push('😀');
549        let res = split_message(&src);
550        assert_eq!(2, res.len());
551        assert_eq!(&src[..MSG_SPLIT_LEN - 1], res[0]);
552        assert_eq!(&src[MSG_SPLIT_LEN - 1..], res[1]);
553    }
