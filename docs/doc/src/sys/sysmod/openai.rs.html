96/// [MODEL_LIST] からモデル名で [OfflineModelInfo] を検索する。
97///
98/// HashMap で検索する。
99fn get_offline_model_info(model: &str) -> Result<&OfflineModelInfo> {
100    static MAP: LazyLock<HashMap<&str, &OfflineModelInfo>> = LazyLock::new(|| {
101        let mut map = HashMap::new();
102        for info in MODEL_LIST.iter() {
103            map.insert(info.name, info);
104        }
105
106        map
107    });
108
109    MAP.get(model)
110        .copied()
111        .ok_or_else(|| anyhow!("Model not found: {model}"))
112}

153/// Rate Limit 情報。
154///
155/// HTTP レスポンスヘッダに含まれる。
156#[derive(Debug, Clone, Copy)]
157struct RateLimit {
158    timestamp: Instant,
159    limit_requests: u32,
160    limit_tokens: u32,
161    remaining_requests: u32,
162    remaining_tokens: u32,
163    reset_requests: Duration,
164    reset_tokens: Duration,
165}
166
167#[derive(Debug, Clone, Copy)]
168pub struct ExpectedRateLimit {
169    pub limit_requests: u32,
170    pub limit_tokens: u32,
171    pub remaining_requests: u32,
172    pub remaining_tokens: u32,
173}
174
175impl RateLimit {
176    fn from(resp: &reqwest::Response) -> Result<Self> {
177        let timestamp = Instant::now();
178        let headers = resp.headers();
179
180        let to_u32 = |key| -> Result<u32> {
181            let s = headers
182                .get(key)
183                .ok_or_else(|| anyhow!("not found: {key}"))?
184                .to_str()?;
185
186            s.parse::<u32>()
187                .with_context(|| format!("parse u32 failed: {s}"))
188        };
189        let to_secs_f64 = |key| -> Result<f64> {
190            let s = headers
191                .get(key)
192                .ok_or_else(|| anyhow!("not found: {key}"))?
193                .to_str()?;
194
195            Self::to_secs_f64(s).with_context(|| format!("parse f64 secs failed: {s}"))
196        };
197
198        let limit_requests = to_u32("x-ratelimit-limit-requests")?;
199        let limit_tokens = to_u32("x-ratelimit-limit-tokens")?;
200        let remaining_requests = to_u32("x-ratelimit-remaining-requests")?;
201        let remaining_tokens = to_u32("x-ratelimit-remaining-tokens")?;
202        let reset_requests = to_secs_f64("x-ratelimit-reset-requests")?;
203        let reset_tokens = to_secs_f64("x-ratelimit-reset-tokens")?;
204
205        Ok(Self {
206            timestamp,
207            limit_requests,
208            limit_tokens,
209            remaining_requests,
210            remaining_tokens,
211            reset_requests: Duration::from_secs_f64(reset_requests),
212            reset_tokens: Duration::from_secs_f64(reset_tokens),
213        })
214    }
215
216    /// サンプルには "6m0s" と書かれているが、
217    /// 実際には "30.828s" のように小数が来ている。
218    /// また、"120ms" とかもある。
219    fn to_secs_f64(s: &str) -> Result<f64> {
220        let mut sum = 0.0;
221
222        let unit_to_scale = |unit: &str| -> Result<f64> {
223            let scale = match unit {
224                "ns" => 0.000_000_001,
225                "us" => 0.000_001,
226                "ms" => 0.001,
227                "s" => 1.0,
228                "m" => 60.0,
229                "h" => 3600.0,
230                "d" => 86400.0,
231                _ => bail!("unknown unit: {unit}"),
232            };
233            Ok(scale)
234        };
235
236        let mut numbuf = String::new();
237        let mut unitbuf = String::new();
238        for c in s.chars() {
239            match c {
240                '0'..='9' | '.' => {
241                    if !unitbuf.is_empty() {
242                        let num = numbuf.parse::<f64>()?;
243                        let scale = unit_to_scale(&unitbuf)?;
244                        sum += num * scale;
245                        numbuf.clear();
246                        unitbuf.clear();
247                    }
248                    numbuf.push(c);
249                }
250                _ => {
251                    unitbuf.push(c);
252                }
253            };
254        }
255        if !unitbuf.is_empty() {
256            let num = numbuf.parse::<f64>()?;
257            let scale = unit_to_scale(&unitbuf)?;
258            sum += num * scale;
259            numbuf.clear();
260            unitbuf.clear();
261        }
262        ensure!(numbuf.is_empty(), "unexpected format: {}", s);
263
264        Ok(sum)
265    }
266
267    fn calc_expected_current(&self) -> ExpectedRateLimit {
268        let now = Instant::now();
269        let elapsed_secs = (now - self.timestamp).as_secs_f64();
270        debug!("{self:?}");
271        debug!("elapsed_secs = {elapsed_secs}");
272
273        let remaining_requests = if elapsed_secs >= self.reset_requests.as_secs_f64() {
274            self.limit_requests
275        } else {
276            let vreq = (self.limit_requests - self.remaining_requests) as f64
277                / self.reset_requests.as_secs_f64();
278            let remaining_requests = self.remaining_requests as f64 + vreq * elapsed_secs;
279
280            (remaining_requests as u32).min(self.limit_requests)
281        };
282
283        let remaining_tokens = if elapsed_secs >= self.reset_tokens.as_secs_f64() {
284            self.limit_tokens
285        } else {
286            let vtok = (self.limit_tokens - self.remaining_tokens) as f64
287                / self.reset_tokens.as_secs_f64();
288            let remaining_tokens = self.remaining_tokens as f64 + vtok * elapsed_secs;
289
290            (remaining_tokens as u32).min(self.limit_tokens)
291        };
292
293        ExpectedRateLimit {
294            limit_requests: self.limit_requests,
295            limit_tokens: self.limit_tokens,
296            remaining_requests,
297            remaining_tokens,
298        }
299    }
300}

618impl OpenAi {
619    /// コンストラクタ。
620    pub fn new() -> Result<Self> {
621        info!("[openai] initialize");
622
623        let config = config::get(|cfg| cfg.openai.clone());
624
625        info!("[openai] OpenAI model list START");
626        for info in MODEL_LIST.iter() {
627            info!(
628                "[openai] name: \"{}\", context_window: {}",
629                info.name, info.context_window
630            );
631        }
632        info!("[openai] OpenAI model list END");
633
634        let info = get_offline_model_info(&config.model)?;
635        info!(
636            "[openai] selected: model: {}, token_limit: {}",
637            info.name, info.context_window
638        );
639
640        if !config.storage_dir.is_empty() {
641            info!("[openai] mkdir: {}", config.storage_dir);
642            std::fs::create_dir_all(&config.storage_dir)?;
643        }
644
645        let client = reqwest::Client::builder()
646            .connect_timeout(CONN_TIMEOUT)
647            .timeout(TIMEOUT)
648            .build()?;
649
650        Ok(OpenAi {
651            config: config.clone(),
652            client,
653            model_name: info.name,
654            model_info_offline: *info,
655            model_info_online: None,
656            rate_limit: None,
657        })
658    }

677    pub async fn model_info_online(&mut self) -> Result<OnlineModelInfo> {
678        let cur = &self.model_info_online;
679        let update = if let Some(info) = cur {
680            let now = SystemTime::now();
681            let elapsed = now.duration_since(info.last_update).unwrap_or_default();
682
683            elapsed > MODEL_INFO_UPDATE_INTERVAL
684        } else {
685            true
686        };
687
688        if update {
689            info!("[openai] update model info");
690            let info = self.get_online_model_info().await?;
691            let info = OnlineModelInfo::from(info);
692            let newval = CachedModelInfo {
693                last_update: SystemTime::now(),
694                info: info.clone(),
695            };
696            let _ = self.model_info_online.insert(newval);
697
698            Ok(info)
699        } else {
700            info!("[openai] skip to update model info");
701            Ok(cur.as_ref().unwrap().info.clone())
702        }
703    }

706    /// 基本的に max_output_tokens に余裕を持たせた値を使うが、
707    /// それが意味をなしていない旧モデルの場合は context_window のうち一定割合とする。
708    pub fn get_output_reserved_token(&self) -> usize {
709        let info = self.model_info_offline();
710        let v1 = (info.max_output_tokens as f32 * MAX_OUTPUT_TOKENS_FACTOR) as usize;
711        let v2 = (info.context_window as f32 * OUTPUT_RESERVED_RATIO) as usize;
712
713        v1.min(v2)
714    }

745    /// エラーチェインの中から特定のエラーを探す。
746    pub fn error_kind(err: &anyhow::Error) -> OpenAiErrorKind {
747        for cause in err.chain() {
748            if let Some(req_err) = cause.downcast_ref::<reqwest::Error>() {
749                if req_err.is_timeout() {
750                    return OpenAiErrorKind::Timeout;
751                }
752            }
753            if let Some(http_err) = cause.downcast_ref::<HttpStatusError>() {
754                // 429: Too Many Requests
755                if http_err.status == 429 {
756                    let msg = http_err.body.to_ascii_lowercase();
757                    if msg.contains("rate") && msg.contains("limit") {
758                        return OpenAiErrorKind::RateLimit;
759                    } else if msg.contains("quota") && msg.contains("billing") {
760                        return OpenAiErrorKind::QuotaExceeded;
761                    }
762                } else {
763                    return OpenAiErrorKind::HttpError(http_err.status);
764                }
765            }
766        }
767
768        OpenAiErrorKind::Fatal
769    }

779    /// JSON を POST して [Response] を返す。
780    /// 成功しても HTTP ステータスコードは失敗かもしれない。
781    ///
782    /// HTTP Header に付与されるメタ情報をログに記録する。
783    /// レートリミット情報は後で参照できるように保存する。
784    ///
785    /// <https://platform.openai.com/docs/api-reference/debugging-requests>
786    async fn post_json(
787        &mut self,
788        url: &str,
789        body: &(impl Serialize + std::fmt::Debug),
790    ) -> Result<Response> {
791        let key = &self.config.api_key;
792
793        info!("[openai] post_json: {url}");
794        info!("[openai] {}", serde_json::to_string(body).unwrap());
795        self.check_enabled()?;
796
797        let resp = self
798            .client
799            .post(url)
800            .header("Authorization", format!("Bearer {key}"))
801            .json(body)
802            .send()
803            .await?;
804        // HTTP POST レスポンス取得まで成功
805        // ステータスコードは失敗かもしれない
806
807        // API メタ情報および Rate Limit 情報をログに記録
808        Self::log_header(&resp, "x-request-id");
809        Self::log_header(&resp, "openai-organization");
810        Self::log_header(&resp, "openai-processing-ms");
811        Self::log_header(&resp, "openai-version");
812        // ドキュメントにはないが、公式ライブラリが使っている
813        // これに従うのがおすすめなのかもしれない
814        Self::log_header(&resp, "x-should-retry");
815
816        // レートリミット情報を最新に更新
817        // 失敗しても警告のみ
818        match RateLimit::from(&resp) {
819            Ok(rate_limit) => {
820                info!("[openai] rate limit: {:?}", rate_limit);
821                self.rate_limit = Some(rate_limit);
822            }
823            Err(err) => {
824                warn!("[openai] could not get rate limit: {err:#}");
825            }
826        }
827
828        Ok(resp)
829    }

866    pub async fn chat(&mut self, msgs: Vec<ChatMessage>) -> Result<String> {
867        info!("[openai] chat request");
868
869        let body = ChatRequest {
870            model: self.model_name.to_string(),
871            messages: msgs,
872            ..Default::default()
873        };
874
875        let json_str = self.post_json_text(URL_CHAT, &body).await?;
876        let resp_msg: ChatResponse = netutil::convert_from_json(&json_str)?;
877
878        // 最初のを選ぶ
879        let choice0 = resp_msg
880            .choices
881            .first()
882            .ok_or(anyhow!("choices is empty"))?;
883        let text = choice0
884            .message
885            .content
886            .as_ref()
887            .ok_or(anyhow!("message content is empty"))?
888            .clone();
889
890        Ok(text)
891    }

968            model,
969            input: input.to_string(),
970            voice,
971            response_format,
972            speed,
973        };
974
975        let bin = self.post_json_bin(URL_AUDIO_SPEECH, &body).await?;
