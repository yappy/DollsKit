127/// 自動削除設定。チャネルごとに保持される。
128#[derive(Clone, Copy)]
129pub struct AutoDeleteConfig {
130    /// 残す数。0 は無効。
131    keep_count: u32,
132    /// 残す時間 (単位は分)。0 は無効。
133    keep_dur_min: u32,
134}

136impl Display for AutoDeleteConfig {
137    /// to_string 可能にする。
138    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
139        let count_str = if self.keep_count != 0 {
140            self.keep_count.to_string()
141        } else {
142            "Disabled".to_string()
143        };
144        let time_str = if self.keep_dur_min != 0 {
145            let (d, h, m) = convert_duration(self.keep_dur_min);
146            format!("{d} day(s) {h} hour(s) {m} minute(s)")
147        } else {
148            "Disabled".to_string()
149        };
150
151        write!(f, "Keep Count: {count_str}\nKeep Time: {time_str}")
152    }
153}

189    async fn init_openai(&mut self, ctrl: &Control) {

202        assert!(chat_history.get_total_limit() == model_info.context_window);
203        let inst_token: usize = self
204            .config
205            .prompt
206            .instructions

208            .map(|text| chat_history.token_count(text))

210        let reserved = FUNCTION_TOKEN + inst_token + reserved;

218        func_table.register_basic_functions();

242    /// 発言を投稿する。
243    ///
244    /// 接続前の場合、接続後まで遅延する。
245    pub async fn say(&mut self, msg: &str) -> Result<()> {
246        if !self.config.enabled {
247            info!("[discord] disabled - msg: {}", msg);
248            return Ok(());
249        }
250        if self.config.notif_channel == 0 {
251            info!("[discord] notification disabled - msg: {}", msg);
252            return Ok(());
253        }
254        if self.ctx.is_none() {
255            info!("[discord] not ready, postponed - msg: {}", msg);
256            self.postponed_msgs.push(msg.to_string());
257            return Ok(());
258        }
259
260        info!("[discord] say msg: {}", msg);
261        let ch = ChannelId::new(self.config.notif_channel);
262        let ctx = self.ctx.as_ref().unwrap();
263        ch.say(ctx, msg).await?;
264
265        Ok(())
266    }
267
268    /// [Self::chat_history] にタイムアウトを適用する。
269    fn check_history_timeout(&mut self) {
270        let now = Instant::now();
271
272        if let Some(timeout) = self.chat_timeout {
273            if now > timeout {
274                self.chat_history_mut().clear();
275                self.chat_timeout = None;
276            }
277        }
278    }

281/// システムを初期化し開始する。
282///
283/// [Discord::on_start] から spawn される。
284async fn discord_main(ctrl: Control) -> Result<()> {
285    let (config, wakeup_list) = {
286        let mut discord = ctrl.sysmods().discord.lock().await;
287        discord.init_openai(&ctrl).await;
288
289        (discord.config.clone(), discord.wakeup_list.clone())
290    };
291
292    // owner_ids を HashSet に変換 (0 は panic するので禁止)
293    let mut owners = HashSet::new();
294    for id in config.owner_ids {
295        if id == 0 {
296            bail!("owner id must not be 0");
297        }
298        owners.insert(UserId::new(id));
299    }
300    info!("[discord] owners: {:?}", owners);
301
302    let ctrl_for_setup = ctrl.clone();
303    let framework = poise::Framework::builder()
304        // owner は手動で設定する
305        .initialize_owners(false)
306        // その他オプション
307        .options(poise::FrameworkOptions {
308            on_error: |err| Box::pin(on_error(err)),
309            pre_command: |ctx| Box::pin(pre_command(ctx)),
310            post_command: |ctx| Box::pin(post_command(ctx)),
311            event_handler: |ctx, ev, fctx, data| Box::pin(event_handler(ctx, ev, fctx, data)),
312            // prefix command
313            prefix_options: poise::PrefixFrameworkOptions {
314                prefix: None,
315                mention_as_prefix: true,
316                ..Default::default()
317            },
318            // owner は手動で設定する (builder の方から設定されるようだがデフォルトが true なので念のためこちらも)
319            initialize_owners: false,
320            owners,
321            // コマンドリスト
322            commands: command_list(),
323            ..Default::default()
324        })
325        // ハンドラ
326        .setup(|ctx, _ready, framework| {
327            // 最初の ready イベントで呼ばれる
328            Box::pin(async move {
329                let mut discord = ctrl_for_setup.sysmods().discord.lock().await;
330                discord.ctx = Some(ctx.clone());
331
332                info!("[discord] register commands...");
333                poise::builtins::register_globally(ctx, &framework.options().commands).await?;
334                info!("[discord] register commands OK");
335
336                // construct user data here (invoked when bot connects to Discord)
337                Ok(PoiseData {
338                    ctrl: Arc::clone(&ctrl_for_setup),
339                })
340            })
341        })
342        .build();
343
344    // クライアントの初期化
345    let intents = GatewayIntents::non_privileged() | GatewayIntents::MESSAGE_CONTENT;
346    let mut client = Client::builder(config.token.clone(), intents)
347        .framework(framework)
348        .await?;
349
350    // システムシャットダウンに対応してクライアントにシャットダウン要求を送る
351    // 別タスクを立ち上げる
352    let ctrl_for_cancel = Arc::clone(&ctrl);
353    let shard_manager = client.shard_manager.clone();
354    taskserver::spawn_oneshot_fn(&ctrl, "discord-exit", async move {
355        ctrl_for_cancel.wait_cancel_rx().await;
356        info!("[discord-exit] recv cancel");
357        shard_manager.shutdown_all().await;
358        info!("[discord-exit] shutdown_all ok");
359        // shutdown_all が完了した後は ready は呼ばれないはずなので
360        // ここで ctx を処分する
361        // ctx.data に Control を持たせているので ctx がリークしたままだと
362        // 終了処理が完了しない
363        let ctx = ctrl_for_cancel.sysmods().discord.lock().await.ctx.take();
364        drop(ctx);
365        info!("[discord-exit] context dropped");
366
367        Ok(())
368    });
369
370    // 定期チェックタスクを立ち上げる
371    taskserver::spawn_periodic_task(&ctrl, "discord-periodic", &wakeup_list, periodic_main);
372
373    // システムスタート
374    client.start().await?;
375    info!("[discord] client exit");
376
377    Ok(())
378}

380/// 文字数制限に気を付けつつ分割して送信する。
381async fn reply_long(ctx: &PoiseContext<'_>, content: &str) -> Result<()> {
382    // mention 関連でのずれが少し怖いので余裕を持たせる
383    const LEN: usize = MSG_MAX_LEN - 128;
384
385    let mut remain = content;
386    loop {
387        let (chunk, fin) = match remain.char_indices().nth(LEN) {
388            Some((ind, _c)) => {
389                let (a, b) = remain.split_at(ind);
390                remain = b;
391
392                (a, false)
393            }
394            None => (remain, true),
395        };
396        if !chunk.is_empty() {
397            ctx.reply(chunk).await?;
398        }
399        if fin {
400            break;
401        }
402    }
403    Ok(())
404}

406/// Markdown エスケープしながら Markdown 引用する。
407/// 文字数制限に気を付けつつ分割して送信する。
408async fn reply_long_mdquote(ctx: &PoiseContext<'_>, content: &str) -> Result<()> {
409    // mention 関連でのずれが少し怖いので余裕を持たせる
410    // 引用符の分も含める
411    const LEN: usize = MSG_MAX_LEN - 128;
412    const QUOTE_PRE: &str = "```\n";
413    const QUOTE_PST: &str = "\n```";
414    const SPECIALS: &str = "\\`";
415
416    let mut count = 0;
417    let mut buf = String::from(QUOTE_PRE);
418    for c in content.chars() {
419        if count >= LEN {
420            buf.push_str(QUOTE_PST);
421            ctx.reply(buf).await?;
422
423            count = 0;
424            buf = String::from(QUOTE_PRE);
425        }
426        if SPECIALS.find(c).is_some() {
427            buf.push('\\');
428        }
429        buf.push(c);
430        count += 1;
431    }
432    if !buf.is_empty() {
433        buf.push_str(QUOTE_PST);
434        ctx.reply(buf).await?;
435    }
436    Ok(())
437}

439/// 分を (日, 時間, 分) に変換する。
440fn convert_duration(mut min: u32) -> (u32, u32, u32) {
441    let day = min / (60 * 24);
442    min %= 60 * 24;
443    let hour = min / 60;
444    min %= 60;
445
446    (day, hour, min)
447}
448
449/// 日時分からなる文字列を分に変換する。
450///
451/// 例: 1d2h3m
452fn parse_duration(src: &str) -> Result<u32> {
453    let mut min = 0u32;
454    let mut buf = String::new();
455    for c in src.chars() {
456        if c == 'd' || c == 'h' || c == 'm' {
457            let n: u32 = buf.parse()?;
458            let n = match c {
459                'd' => n.saturating_mul(24 * 60),
460                'h' => n.saturating_mul(60),
461                'm' => n,
462                _ => panic!(),
463            };
464            min = min.saturating_add(n);
465            buf.clear();
466        } else {
467            buf.push(c);
468        }
469    }
470    Ok(min)
471}

473/// チャネル内の全メッセージを取得し、フィルタ関数が true を返したものを
474/// すべて削除する。
475///
476/// Bulk delete 機能で一気に複数を消せるが、2週間以上前のメッセージが
477/// 含まれていると BAD REQUEST になる等扱いが難しいので rate limit は
478/// 気になるが1つずつ消す。
479///
480/// * `ctx` - HTTP コンテキスト。
481/// * `ch` - Channel ID。
482/// * `filter` - (メッセージ, 番号, 総数) から消すならば true を返す関数。
483///
484/// (消した数, 総メッセージ数) を返す。
485async fn delete_msgs_in_channel<F: Fn(&Message, usize, usize) -> bool>(
486    ctx: &Context,
487    ch: ChannelId,
488    filter: F,
489) -> Result<(usize, usize)> {
490    // id=0 から 100 件ずつすべてのメッセージを取得する
491    let mut allmsgs = BTreeMap::<MessageId, Message>::new();
492    const GET_MSG_LIMIT: u8 = 100;
493    let mut after = None;
494    loop {
495        // https://discord.com/developers/docs/resources/channel#get-channel-messages
496        info!("get_messages: after={:?}", after);
497        let target = after.map(MessagePagination::After);
498        let msgs = ctx
499            .http
500            .get_messages(ch, target, Some(GET_MSG_LIMIT))
501            .await?;
502        // 空配列ならば完了
503        if msgs.is_empty() {
504            break;
505        }
506        // 降順で送られてくるので ID でソートし直す
507        allmsgs.extend(msgs.iter().map(|m| (m.id, m.clone())));
508        // 最後の message id を次回の after に設定する
509        after = Some(*allmsgs.keys().next_back().unwrap());
510    }
511    info!("obtained {} messages", allmsgs.len());
512
513    let mut delcount = 0_usize;
514    for (i, (&mid, msg)) in allmsgs.iter().enumerate() {
515        if !filter(msg, i, allmsgs.len()) {
516            continue;
517        }
518        // ch, msg ID はログに残す
519        info!("Delete: ch={}, msg={}", ch, mid);
520        // https://discord.com/developers/docs/resources/channel#delete-message
521        ctx.http.delete_message(ch, mid, None).await?;
522        delcount += 1;
523    }
524    info!("deleted {} messages", delcount);
525
526    Ok((delcount, allmsgs.len()))
527}

529/// 自動削除周期タスク。
530async fn periodic_main(ctrl: Control) -> Result<()> {
531    let (ctx, config_map) = {
532        let discord = ctrl.sysmods().discord.lock().await;
533        if discord.ctx.is_none() {
534            // ready 前なら何もせず正常終了する
535            return Ok(());
536        }
537        (
538            discord.ctx.as_ref().unwrap().clone(),
539            discord.auto_del_config.clone(),
540        )
541    };
542
543    // UNIX timestamp [sec]
544    let now = Utc::now().timestamp() as u64;
545
546    for (ch, config) in config_map {
547        let AutoDeleteConfig {
548            keep_count,
549            keep_dur_min,
550        } = config;
551        if keep_count == 0 && keep_dur_min == 0 {
552            continue;
553        }
554        let keep_dur_sec = (keep_dur_min as u64).saturating_mul(60);
555        let (_delcount, _total) = delete_msgs_in_channel(&ctx, ch, |msg, i, len| {
556            let mut keep = true;
557            if keep_count != 0 {
558                keep = keep && i + (keep_count as usize) < len;
559            }
560            if keep_dur_min != 0 {
561                let created = msg.timestamp.timestamp() as u64;
562                // u64 [sec] 同士の減算で経過時間を計算する
563                // オーバーフローは代わりに 0 とする
564                let duration = now.saturating_sub(created);
565                keep = keep && duration <= keep_dur_sec;
566            }
567            !keep
568        })
569        .await?;
570    }
571
572    Ok(())
573}

834/// AI assistant.
835///
836/// The owner of the assistant will pay the usage fee for ChatGPT.
837#[poise::command(slash_command, prefix_command, category = "AI")]
838async fn ai(
839    ctx: PoiseContext<'_>,
840    #[description = "Chat message to AI assistant"]
841    #[min_length = 1]
842    #[max_length = 1024]
843    chat_msg: String,
844    #[description = "Show internal details when AI calls a function. (default=False)"]
845    trace_function_call: Option<bool>,
846    web_search_quality: Option<WebSearchQuality>,
847) -> Result<(), PoiseError> {
848    // そのまま引用返信
849    reply_long_mdquote(&ctx, &chat_msg).await?;
850
851    let data = ctx.data();
852    let mut discord = data.ctrl.sysmods().discord.lock().await;
853
854    // タイムアウト処理
855    discord.check_history_timeout();
856
857    // 今回の発言をヒストリに追加 (システムメッセージ + 本体)
858    let sysmsg = discord
859        .config
860        .prompt
861        .each
862        .join("")
863        .replace("${user}", &ctx.author().name);
864    discord
865        .chat_history_mut()
866        .push_message(Role::Developer, &sysmsg)?;
867    discord
868        .chat_history_mut()
869        .push_message(Role::User, &chat_msg)?;
870
871    // システムメッセージ
872    let inst = discord.config.prompt.instructions.join("");
873    // ツール (function + built-in tools)
874    let mut tools = vec![];
875    // web search
876    let web_csize = match web_search_quality.unwrap_or_default() {
877        WebSearchQuality::High => Some(SearchContextSize::High),
878        WebSearchQuality::Medium => Some(SearchContextSize::Medium),
879        WebSearchQuality::Low => Some(SearchContextSize::Low),
880        WebSearchQuality::Disabled => None,
881    };
882    if let Some(web_csize) = web_csize {
883        tools.push(Tool::WebSearchPreview {
884            search_context_size: Some(web_csize),
885            user_location: Some(UserLocation::default()),
886        });
887    }
888    // function
889    for f in discord.func_table().function_list() {
890        tools.push(Tool::Function(f.clone()));
891    }
892
893    // AI 返答まで関数呼び出しを繰り返す
894    let result = loop {
895        // 入力をヒストリの内容から作成
896        let input = Vec::from_iter(discord.chat_history().iter().cloned());
897        // ChatGPT API
898        let resp = {
899            let mut ai = data.ctrl.sysmods().openai.lock().await;
900            ai.chat_with_tools(Some(&inst), input, &tools).await
901        };
902        match resp {
903            Ok(resp) => {
904                // function 呼び出しがあれば履歴に追加
905                for fc in resp.func_call_iter() {
906                    let call_id = &fc.call_id;
907                    let func_name = &fc.name;
908                    let func_args = &fc.arguments;
909
910                    // call function
911                    let func_out = discord.func_table().call((), func_name, func_args).await;
912                    // debug trace
913                    if discord.func_table.as_ref().unwrap().debug_mode()
914                        || trace_function_call.unwrap_or(false)
915                    {
916                        reply_long(
917                            &ctx,
918                            &format!(
919                                "function call: {func_name}\nparameters: {func_args}\nresult: {}",
920                                func_out
921                            ),
922                        )
923                        .await?;
924                    }
925                    // function の結果を履歴に追加
926                    discord
927                        .chat_history_mut()
928                        .push_function(call_id, func_name, func_args, &func_out)?;
929                }
930                // アシスタント応答と web search があれば履歴に追加
931                let text = resp.output_text();
932                if !text.is_empty() {
933                    discord.chat_history_mut().push_message_tool(
934                        std::iter::once((Role::Assistant, text.clone())),
935                        resp.web_search_iter().cloned(),
936                    )?;
937                } else {
938                    discord
939                        .chat_history_mut()
940                        .push_message_tool(std::iter::empty(), resp.web_search_iter().cloned())?;
941                }
942
943                if !text.is_empty() {
944                    break Ok(text);
945                }
946            }
947            Err(err) => {
948                // エラーが発生した
949                error!("{:#?}", err);
950                break Err(err);
951            }
952        }
953    };
954
955    // discord 返信
956    match result {
957        Ok(reply_msg) => {
958            info!("[discord] openai reply: {reply_msg}");
959            reply_long(&ctx, &reply_msg).await?;
960
961            // タイムアウト延長
962            discord.chat_timeout = Some(
963                Instant::now()
964                    + Duration::from_secs(discord.config.prompt.history_timeout_min as u64 * 60),
965            );
966        }
967        Err(err) => {
968            error!("[discord] openai error: {:#?}", err);
969            let errmsg = match OpenAi::error_kind(&err) {
970                OpenAiErrorKind::Timeout => "Server timed out.".to_string(),
971                OpenAiErrorKind::RateLimit => {
972                    "Rate limit exceeded. Please retry after a while.".to_string()
973                }
974                OpenAiErrorKind::QuotaExceeded => "Quota exceeded. Charge the credit.".to_string(),
975                OpenAiErrorKind::HttpError(status) => format!("Error {status}"),
976                _ => "Error".to_string(),
977            };
978
979            warn!("[discord] openai reply: {errmsg}");
980            ctx.reply(errmsg).await?;
981        }
982    }
983
984    Ok(())
985}

1186/// Poise イベントハンドラ。
1187///
1188/// [poise::builtins::on_error] のままでまずい部分を自分でやる。
1189async fn on_error(error: poise::FrameworkError<'_, PoiseData, PoiseError>) {
1190    // エラーを返していないはずのものは panic にする
1191    match error {
1192        poise::FrameworkError::Setup { error, .. } => {
1193            panic!("Failed on setup: {:#?}", error)
1194        }
1195        poise::FrameworkError::EventHandler { error, .. } => {
1196            panic!("Failed on eventhandler: {:#?}", error)
1197        }
1198        poise::FrameworkError::Command { error, ctx, .. } => {
1199            error!(
1200                "[discord] error in command `{}`: {:#?}",
1201                ctx.command().name,
1202                error
1203            );
1204        }
1205        poise::FrameworkError::NotAnOwner { ctx, .. } => {
1206            let errmsg = ctx
1207                .data()
1208                .ctrl
1209                .sysmods()
1210                .discord
1211                .lock()
1212                .await
1213                .config
1214                .perm_err_msg
1215                .clone();
1216            info!("[discord] not an owner: {}", ctx.author());
1217            info!("[discord] reply: {errmsg}");
1218            if let Err(why) = ctx.reply(errmsg).await {
1219                error!("[discord] reply error: {:#?}", why)
1220            }
1221        }
1222        poise::FrameworkError::UnknownInteraction { interaction, .. } => {
1223            warn!(
1224                "[discord] received unknown interaction \"{}\"",
1225                interaction.data.name
1226            );
1227        }
1228        error => {
1229            if let Err(why) = poise::builtins::on_error(error).await {
1230                error!("[discord] error while handling error: {:#?}", why)
1231            }
1232        }
1233    }
1234}

1236/// Serenity の全イベントハンドラ。
1237///
1238/// Poise のコンテキストが渡されるので、Serenity ではなく Poise の
1239/// FrameworkOptions 経由で設定する。
1240async fn event_handler(
1241    ctx: &Context,
1242    ev: &FullEvent,
1243    _fctx: FrameworkContext<'_, PoiseData, PoiseError>,
1244    data: &PoiseData,
1245) -> Result<(), PoiseError> {
1246    match ev {
1247        FullEvent::Ready { data_about_bot } => {
1248            info!("[discord] connected as {}", data_about_bot.user.name);
1249            Ok(())
1250        }
1251        FullEvent::Resume { event: _ } => {
1252            info!("[discord] resumed");
1253            Ok(())
1254        }
1255        FullEvent::CacheReady { guilds } => {
1256            // このタイミングで [Discord::ctx] に ctx をクローンして保存する。
1257            info!("[discord] cache ready - guild: {}", guilds.len());
1258
1259            let mut discord = data.ctrl.sysmods().discord.lock().await;
1260            let ctx_clone = ctx.clone();
1261            discord.ctx = Some(ctx_clone);
1262
1263            info!(
1264                "[discord] send postponed msgs ({})",
1265                discord.postponed_msgs.len()
1266            );
1267            for msg in &discord.postponed_msgs {
1268                let ch = discord.config.notif_channel;
1269                // notif_channel が有効の場合しかキューされない
1270                assert_ne!(0, ch);
1271
1272                info!("[discord] say msg: {}", msg);
1273                let ch = ChannelId::new(ch);
1274                if let Err(why) = ch.say(&ctx, msg).await {
1275                    error!("{:#?}", why);
1276                }
1277            }
1278            discord.postponed_msgs.clear();
1279            Ok(())
1280        }
1281        _ => Ok(()),
1282    }
1283}
